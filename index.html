if (!remote) { setStatus('☁ Ready', '#4ade80'); refreshUI(); return; } // Merge quiz history if (Array.isArray(remote.quizHistory) && remote.quizHistory.length) { const local = JSON.parse(localStorage.getItem('quizHistory') || '[]'); const keys = new Set(local.map(h => h.date + '|' + h.name)); const add = remote.quizHistory.filter(h => !keys.has(h.date + '|' + h.name)); if (add.length) { localStorage.setItem('quizHistory', JSON.stringify([...local,...add].sort((a,b)=>b.date.localeCompare(a.date)))); } } // Merge question bank if (Array.isArray(remote.questionBank) && remote.questionBank.length) { const local = JSON.parse(localStorage.getItem('mockiq_question_bank') || '[]'); const ids = new Set(local.map(q => q.id)); const add = remote.questionBank.filter(q => !ids.has(q.id)); if (add.length) { localStorage.setItem('mockiq_question_bank', JSON.stringify([...local,...add])); } } // Merge review questions if (Array.isArray(remote.reviewQuestions) && remote.reviewQuestions.length) { const local = JSON.parse(localStorage.getItem('rm_review_questions') || '[]'); const ids = new Set(local.map(q => q.id || q.question)); const add = remote.reviewQuestions.filter(q => !ids.has(q.id || q.question)); if (add.length) { localStorage.setItem('rm_review_questions', JSON.stringify([...local,...add])); } } // Restore saved mid-mock if (remote.savedMock && !localStorage.getItem('rm_saved_mock')) { localStorage.setItem('rm_saved_mock', JSON.stringify(remote.savedMock)); } const t = new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'}); setStatus('☁ ' + t, '#4ade80'); // ALWAYS refresh UI after pull — not just when new data arrived refreshUI(); } catch(e) { setStatus(e.name === 'AbortError' ? '⏱ Timeout' : '⚠ Error', '#f87171'); console.warn('fbPull:', e.message); } }; // ── SYNC — click button does PULL first then PUSH ──────── // This is what the status indicator calls on click window.fbSync = async function() { await window.fbPull(); // get latest from Firebase first await window.fbPush(); // then push any local changes }; // Pull on page load window.addEventListener('load', () => setTimeout(window.fbPull, 300)); // Back online: full sync window.addEventListener('online', async () => { setStatus('↑ Reconnecting...', 'var(--text-muted)'); await window.fbPull(); await window.fbPush(); }); window.addEventListener('offline', () => setStatus('📵 Offline', 'var(--text-muted)')); })();
